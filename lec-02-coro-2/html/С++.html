Example 1

generator<int> numbers() {
    int i = 1;
    while (true) {
        co_yield i;
        i++;
    }
}

template <class Predicate>
auto filter(Predicate pred) {
    return [pred](generator<int>& gen) -> generator<int> {
        for (int i : gen) {
            if (pred(i)) co_yield i;
        }
    };
}

template <class Functor>
auto transform(Functor functor) {
    return [functor](generator<int>& gen) -> generator<int> {
        for (int i : gen) {
            co_yield functor(i);
        }
    };
}

Example 1

auto take(size_t n) {
    return [n](generator<int>& gen) -> generator<int> {
        size_t taken = 0;
        for (int i : gen) {
            co_yield i;
            taken++;
            if (taken == n) break;
        }
    };
}

int sum(generator<int> gen) {
    int s = 0;
    for (int i : gen) {
        s += i;
    }
    return s;
}

template <class T>
struct Node {
    T value;
    Node* left = nullptr;
    Node* right = nullptr;
};

template <class T>
generator<T> flatten(const Node<T>* root) {
    if (!root) co_return;
    for (const auto& v : flatten(root->left)) co_yield v;
    co_yield root->value;
    for (const auto& v : flatten(root->right)) co_yield v;
}

for (const int& v : flatten(create_some_tree())) {
    std::cout << v << " ";
}

Echo server

struct WriteAwaiter {
    bool await_ready() { return false; }

    void await_suspend(std::coroutine_handle<> h) {
        asio::async_write(
            socket_, buffer_,
            [this, h](int err, size_t) mutable {
                err_ = err;
                h.resume
            });
    }

    void await_resume() {
        if (err_) {
            throw std::system_error(err_);
        }
    }

    tcp::socket& socket_;
    asio::const_buffer buffer_;

    std::error_code err_;
};

auto async_write(tcp::socket& socket, asio::const_buffer buffer) {
    return WriteAwaiter{socket, buffer};
}

Using ThreadPool

auto operator co_await(ThreadPool& tp) {
    struct awaiter {
        ThreadPool& tp;
        bool await_ready() { return false; }
        void await_suspend(std::coroutine_handle<> h) {
            pool.submit([h] { h.resume });
        }
        void await_resume() { }
    };

    return awaiter{ tp };
}

std::future as ResultType

template <class T>
struct StdFutureCoroPromise {
    std::promise<T> promise_;
    auto get_return_object() { return promise_.get_future(); }
    std::suspend_never initial_suspend() noexcept { return {}; }
    std::suspend_never final_suspend() noexcept { return {}; }
    void unhandled_exception() { promise_.set_exception(std::current_exception()); }
    void return_value(T t) { promise_.set_value(value); }
};

template <class T, class... Args>
struct std::coroutine_traits<std::future<R>, Args...> {
    using promise_type = StdFutureCoroPromise<R>;
};

Coroutine compositions

std::future<int> Coro2() {
    co_return 42;
}

std::future<int> Coro1() {
    auto a = co_await Coro2();
    co_return *a * 2;
}

std::future<int> Coro() {
    std::cout << "step 1" << std::endl;
    auto c1 = co_await Coro1();
    std::cout << "step 2" << std::endl;
    auto c2 = co_await Coro2();
    std::cout << "done" << std::endl;
    co_return *c1 + *c2;
}
